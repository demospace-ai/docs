<Tab title="Vue">

First, install the Fabra Connect SDK in your frontend:

```sh
npm install --save @fabra/vue-fabra-connect
```

In your frontend, initialize Fabra Connect on the page where you want to render the component:

```markup
<script>
import FabraConnect from "@fabra/vue-fabra-connect";
export default {
  components: { FabraConnect },
};
</script>
<template>
  <fabra-connect :linkToken="ADD_GENERATED_LINK_TOKEN">
    <template #button="props">
      <button @click="props.onClick">Open Fabra Connect</button>
    </template>
  </fabra-connect>
</template>
```
